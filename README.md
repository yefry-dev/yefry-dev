```
🔭  Telescopio Vera Rubin   →  15 TB de datos por noche  →  Solo la IA puede procesarlos
🌊  LIGO (Ondas Gravitac.)  →  Ruido cuántico vs señal   →  CNNs detectan colisiones de agujeros negros  
🪐  Misión Kepler / TESS    →  Curvas de luz estelares   →  ML descubre exoplanetas automáticamente
🌀  Event Horizon Telescope →  Fotografiar agujeros negros→  Algoritmos de imagen reconstruyen la foto
🌌  Galaxy Zoo              →  Millones de galaxias       →  Deep Learning clasifica morfología galáctica
```

```python
# 🌌 El proyecto que quiero construir algún día
import torch
import torch.nn as nn
from astropy.io import fits
from astropy.visualization import ZScaleInterval

class CosmicVisionNet(nn.Module):
    """
    Red neuronal para clasificar morfología galáctica
    Entrenada con datos reales del Sloan Digital Sky Survey (SDSS)
    """
    def __init__(self, num_clases=5):
        super().__init__()
        # Espiral · Elíptica · Irregular · En fusión · Anillo
        self.clases = ["🌀 Espiral", "⭕ Elíptica",
                       "💫 Irregular", "🔥 Fusión", "💍 Anillo"]
        self.backbone = nn.Sequential(
            nn.Conv2d(1, 32, kernel_size=3, padding=1),
            nn.ReLU(),
            nn.MaxPool2d(2),
            nn.Conv2d(32, 64, kernel_size=3, padding=1),
            nn.ReLU(),
            nn.AdaptiveAvgPool2d((8, 8)),
        )
        self.classifier = nn.Linear(64 * 8 * 8, num_clases)

    def forward(self, imagen_fits):
        features = self.backbone(imagen_fits)
        return self.classifier(features.flatten(1))
```
